```javascript
import { kv, readBody } from './_shared.js';
import { nanoid } from 'nanoid';

export default async function handler(req, res) {
  const id = new URL(req.url, 'http://x').searchParams.get('id');

  // GET list or single
  if (req.method === 'GET') {
    if (id) {
      const raw = await kv.get(`signal:${id}`);
      if (!raw) return res.status(404).json({ error: 'not found' });
      return res.json(JSON.parse(raw));
    }
    const ids = JSON.parse(await kv.get('signals:index') || '[]');
    const pipe = kv.pipeline();
    ids.forEach(i => pipe.get(`signal:${i}`));
    const rows = (await pipe.exec()).map(r => r[1]).filter(Boolean).map(r => JSON.parse(r));
    rows.sort((a, b) => (b.createdAt || '').localeCompare(a.createdAt || ''));
    return res.json(rows);
  }

  // POST create
  if (req.method === 'POST') {
    const body = await readBody(req);
    const sig = {
      id: nanoid(8),
      title: body.title || '',
      summary: body.summary || '',
      source: body.source || '',
      url: body.url || null,
      channel: body.channel || 'manual',
      tags: body.tags || [],
      strength: body.strength ?? 0.5,
      dim: body.dim || null,
      trendId: body.trendId || null,
      campaignId: body.campaignId || null,
      clusterIds: body.clusterIds || [],
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    };
    const ids = JSON.parse(await kv.get('signals:index') || '[]');
    ids.push(sig.id);
    await kv.set('signals:index', JSON.stringify(ids));
    await kv.set(`signal:${sig.id}`, JSON.stringify(sig));
    return res.status(201).json(sig);
  }

  // PUT update
  if (req.method === 'PUT' && id) {
    const raw = await kv.get(`signal:${id}`);
    if (!raw) return res.status(404).json({ error: 'not found' });
    const existing = JSON.parse(raw);
    const body = await readBody(req);
    const updated = { ...existing, ...body, id, updatedAt: new Date().toISOString() };
    await kv.set(`signal:${id}`, JSON.stringify(updated));
    return res.json(updated);
  }

  // DELETE
  if (req.method === 'DELETE' && id) {
    const ids = JSON.parse(await kv.get('signals:index') || '[]');
    await kv.set('signals:index', JSON.stringify(ids.filter(i => i !== id)));
    await kv.del(`signal:${id}`);
    return res.json({ ok: true });
  }

  res.status(405).json({ error: 'method not allowed' });
}
```

import { Icon, StageBadge, DimensionDot } from './ui.jsx';

export const SignalList = ({ t, onOpenTrend }) => {
  const [signals, setSignals] = useState([]);
  const [loading, setLoading] = useState(true);
  const [filter, setFilter] = useState({ channel: '', assigned: '' });

    signalsApi.list().then(s => { setSignals(s); setLoading(false); }).catch(() => setLoading(false));

  const filtered = useMemo(() => {

    if (filter.channel) s = s.filter(x => x.channel === filter.channel);
    if (filter.assigned === 'yes') s = s.filter(x => x.trendId || x.clusterIds.length > 0);
    if (filter.assigned === 'no') s = s.filter(x => !x.trendId && x.clusterIds.length === 0);
    return s;
  }, [signals, filter]);

  const channels = ['manual', 'url', 'pdf', 'ai-scout'];

  if (loading) return <div style={{ padding: 32, color: 'var(--fg-3)' }}>Loading signals…</div>;

  return (
    <div style={{ padding: '0 24px 24px' }}>
      <div style={{ display: 'flex', alignItems: 'center', gap: 12, marginBottom: 16 }}>
        <h2 style={{ margin: 0, fontSize: 18, fontWeight: 600, color: 'var(--fg-0)', flex: 1 }}>

        </h2>
        <select className="btn sm" value={filter.channel} onChange={e => setFilter(f => ({ ...f, channel: e.target.value }))} style={{ fontSize: 12 }}>
          <option value="">All channels</option>
          {channels.map(c => <option key={c} value={c}>{c}</option>)}

          <option value="">All</option>
          <option value="yes">Assigned</option>
          <option value="no">Unassigned</option>
        </select>
      </div>

      {filtered.length === 0 && (
        <div className="card" style={{ padding: 32, textAlign: 'center', color: 'var(--fg-3)' }}>
          No signals yet. Capture your first signal to get started.
        </div>
      )}

      <div style={{ display: 'flex', flexDirection: 'column', gap: 6 }}>
        {filtered.map(sig => (
          <div key={sig.id} className="card" style={{ padding: '10px 14px', display: 'flex', alignItems: 'center', gap: 12 }}>
            <div style={{ width: 8, height: 8, borderRadius: '50%', background: sig.channel === 'ai-scout' ? 'var(--ai)' : sig.channel === 'url' ? 'var(--accent)' : 'var(--fg-3)', flexShrink: 0 }}/>

                {sig.tags.slice(0, 3).map(tag => <span key={tag} className="chip sm">{tag}</span>)}

            <div style={{ fontSize: 11, color: 'var(--fg-3)', whiteSpace: 'nowrap' }}>
              {sig.clusterIds.length > 0 ? `${sig.clusterIds.length} cluster` : sig.trendId ? 'linked' : '—'}

const NewCampaignDialog = ({ open, onClose, onCreated }) => {
  const [form, setForm] = useState({ title: '', description: '', question: '', owner: '', tags: '' });
  const [saving, setSaving] = useState(false);
  const set = (k, v) => setForm(f => ({ ...f, [k]: v }));

  if (!open) return null;

  const handleSave = async () => {
    if (!form.title.trim()) return;
    setSaving(true);
    try {
      const campaign = await campaignsApi.create({
        ...form,
        tags: form.tags ? form.tags.split(',').map(t => t.trim()).filter(Boolean) : [],
      });
      onCreated(campaign);
      onClose();
      setForm({ title: '', description: '', question: '', owner: '', tags: '' });
    } catch (e) {
      alert('Error creating campaign: ' + e.message);
    } finally {
      setSaving(false);
    }
  };

  return (
    <div style={{ position: 'fixed', inset: 0, background: 'rgba(0,0,0,.55)', display: 'flex', alignItems: 'center', justifyContent: 'center', zIndex: 900 }} onClick={onClose}>
      <div className="card" style={{ width: 480, padding: 24 }} onClick={e => e.stopPropagation()}>

Create `api/clusters.js`:

```javascript
import { kv, readBody } from './_shared.js';
import { nanoid } from 'nanoid';

export default async function handler(req, res) {
  const id = new URL(req.url, 'http://x').searchParams.get('id');

  if (req.method === 'GET') {
    if (id) {
      const raw = await kv.get(`cluster:${id}`);
      if (!raw) return res.status(404).json({ error: 'not found' });
      return res.json(JSON.parse(raw));
    }
    const ids = JSON.parse(await kv.get('clusters:index') || '[]');
    const pipe = kv.pipeline();
    ids.forEach(i => pipe.get(`cluster:${i}`));
    const rows = (await pipe.exec()).map(r => r[1]).filter(Boolean).map(r => JSON.parse(r));
    rows.sort((a, b) => (b.createdAt || '').localeCompare(a.createdAt || ''));
    return res.json(rows);
  }

  if (req.method === 'POST') {
    const body = await readBody(req);
    const cluster = {
      id: nanoid(8),
      label: body.label || '',
      description: body.description || '',
      color: body.color || '#A78BFA',
      confidence: body.confidence ?? null,
      signalIds: body.signalIds || [],
      campaignId: body.campaignId || null,
      origin: body.origin || 'manual',
      proposed: false,
      createdAt: new Date().toISOString(),
      updatedAt: new Date().toISOString(),
    };
    const ids = JSON.parse(await kv.get('clusters:index') || '[]');
    ids.push(cluster.id);
    await kv.set('clusters:index', JSON.stringify(ids));
    await kv.set(`cluster:${cluster.id}`, JSON.stringify(cluster));
    return res.status(201).json(cluster);
  }

  if (req.method === 'PUT' && id) {
    const raw = await kv.get(`cluster:${id}`);
    if (!raw) return res.status(404).json({ error: 'not found' });
    const existing = JSON.parse(raw);
    const body = await readBody(req);
    const updated = { ...existing, ...body, id, updatedAt: new Date().toISOString() };
    await kv.set(`cluster:${id}`, JSON.stringify(updated));
    return res.json(updated);
  }

  if (req.method === 'DELETE' && id) {
    const ids = JSON.parse(await kv.get('clusters:index') || '[]');
    await kv.set('clusters:index', JSON.stringify(ids.filter(i => i !== id)));
    await kv.del(`cluster:${id}`);
    return res.json({ ok: true });
  }

  res.status(405).json({ error: 'method not allowed' });
}
```
